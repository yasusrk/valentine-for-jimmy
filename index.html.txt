    <div class="container">
        <div id="gifContainer">

        </div>

        <h1 id="question">Will you be my Valentine? ‚ù§Ô∏è</h1>

        <div class="btn-container">
            <button id="yesBtn" onclick="handleYes()">Yes</button>
            <button id="noBtn" onclick="handleNo()">No</button>
        </div>
    </div>

    <script>
        let noCount = 0;
        const questionText = document.getElementById('question');
        const mainGif = document.getElementById('mainGif');
        const noBtn = document.getElementById('noBtn');
        const yesBtn = document.getElementById('yesBtn');

        // Aapki sequence ke according list
        const noGifs = [
            "dudu-surprise.gif",             // 1st No
            "dudu-wearing-glasses-standing.gif", // 2nd No
            "hi.gif",                        // 3rd No
            "cry.gif",                       // 4th No
            "cry2.gif"                       // 5th No
        ];

        const messages = [
            "Soch lo... surprise hai!", 
            "Swag toh dekho, 'No' kaise bol rahe ho?", 
            "Ek baar aur soch lo? Please?", 
            "Ab rona aa raha hai mujhe...", 
            "Last chance... rula hi diya na? üò≠"
        ];

        function handleNo() {
            if (noCount < noGifs.length) {
                // Change GIF and Text
                mainGif.src = noGifs[noCount];
                questionText.innerHTML = messages[noCount];
                
                // Yes button gets bigger!
                let currentSize = parseFloat(window.getComputedStyle(yesBtn).fontSize);
                yesBtn.style.fontSize = (currentSize + 15) + "px";
                yesBtn.style.padding = (currentSize + 5) + "px " + (currentSize + 25) + "px";
                
                noCount++;
            }

            // Agar 5 baar 'No' click ho jaye toh button gayab ya disable
            if (noCount === noGifs.length) {
                noBtn.style.display = 'none';
                questionText.innerHTML = "Ab toh 'Yes' hi bolna padega! üò§";
            }
        }

        function handleYes() {
            questionText.innerHTML = "I know, you love me soo much! ‚ù§Ô∏è";
            mainGif.src = "kiss.gif"; // Last vala GIF
            noBtn.style.display = 'none';
            yesBtn.style.display = 'none';
            
            // Background confetti effect ki feeling ke liye color change
            document.body.style.backgroundColor = "#ff4d6d";
            questionText.style.color = "white";
        }
    </script>
